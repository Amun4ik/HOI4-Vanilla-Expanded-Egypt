	id = egypt.2
	title = egypt.2.t
	desc = egypt.2.d
	picture = GFX_report_event_generic_funeral

	fire_only_once = yes
	
	trigger = {
		original_tag = EGY
		date > 1936.4.1
		has_country_leader = {
			character = EGY_Faud_i
			ruling_only = yes 
		}
	}
	
	immediate = {
		hidden_effect = {
            retire_character = EGY_Faud_i
			add_country_leader_role = {
				character =  EGY_Farouk_Muhammad_Ali
				promote_leader = yes
				country_leader = {
					ideology = despotism
					expire = "1965.1.1.1"
					traits = { EGY_hedonist EGY_underage_monarch }
				}
			}
		}
	}

	option = {
		name = egypt.2.a
		
		effect_tooltip = {
            retire_character = EGY_Faud_i
			add_country_leader_role = {
				character =  EGY_Farouk_Muhammad_Ali
				promote_leader = yes
				country_leader = {
					ideology = despotism
					expire = "1965.1.1.1"
					traits = { EGY_hedonist EGY_underage_monarch }
				}
			}
		}
	}
